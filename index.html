349. 两个数组的交集

代码

class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        int[] arr = new int[1000];
        for (int i : nums1) {
            arr[i]++;
        }
        Set<Integer> set = new HashSet();
        for (int j : nums2) {
            if (arr[j] > 0) {
                set.add(j);
            }
        }
        int[] result = new int[set.size()];
        int i = 0;
        for (int r : set) {
            result[i++] = r;
        }
        return result;
    }
}

242. 有效的字母异位词

代码

class Solution {
    public boolean isAnagram(String s, String t) {
        int[] arr = new int[26];
        for(char c:s.toCharArray()){
            arr[c-'a']++;
        }
        for(char c:t.toCharArray()){
            arr[c-'a']--;
        }
        for(int i=0;i<arr.length;i++){
            if(arr[i] != 0){
                return false;
            }
        }
        return true;
    }
}
## 总结
最后判结果时，不能累加判和，需要逐位判断
